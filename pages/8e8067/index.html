#!/bin/bash
  
harbor_url=$1
harbor_project_name=$2
project_name=$3
tag=$4
port=$5
harbor_username=$6
harbor_password=$7

image_name="${harbor_url}/${harbor_project_name}/${project_name}:${tag}"


containerId=`docker ps -a | grep -w ${project_name}:${tag} | awk '{print $1}'`

if [ "$containerId" != "" ]; then
    docker stop $containerId
    docker rm $containerId
    echo "Container deleted successfully"
fi

docker login -u ${harbor_username} -p ${harbor_password} ${harbor_url}

docker pull $image_name

docker run -itd -p ${port}:${port} --name ${project_name} ${image_name}

docker rmi $(docker images -q -f dangling=true)

echo "Deployment complete"
